CREATE TABLE Customer
(
    CNM int(200) PRIMARY KEY,
    CNAME varchar(200),
    CITY varchar(200),
    RATING int(200),
    SNo int(200),
    FOREIGN KEY (SNo) REFERENCES salesperson(SNo)
);
